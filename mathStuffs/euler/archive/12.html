Drawing Flower

Design an algorithm to draw a flower.

Javascript

var canvas;
var ctx;
var cvsSize;
var maxSize;
// set up necessary functions initialize these variables

var halfSize = cvsSize/2;
var gr = 1.61803398875;
var flcol = [100,35,0];
var overlaycol = [200,200,200];

function drawPetal(n, size){
    ctx.beginPath();
	ctx.rotate(Math.PI*2/gr);
    ctx.moveTo(0,0);
    ctx.bezierCurveTo(size*0.4,size*0.5,size*0.4,size*0.8,0,size);
    ctx.bezierCurveTo(-size*0.4,size*0.8,-size*0.4,size*0.5,0,0);
    ctx.fill();
    ctx.moveTo(0,0);
}

function drawFlower(n,c){
	if (n>54) return;
	ctx.globalAlpha = 0.9
	ctx.fillStyle = "rgb("+Math.min(c[0]+n*10,225).toString()+
					","+Math.min(c[1]+n*10,225).toString()+
					","+Math.min(c[2]+n*10,225).toString()+")";
	s = Math.max(maxSize-n*3,0);
	if (s==0) return;
	drawPetal(n,s);
	setTimeout(function(){ drawFlower(n+1,c); }, 110);
}
