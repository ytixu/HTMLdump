library(segmented)

N = 1000

nSize = c(5, 10, 20, 50)

sigmas = c(1)

fun = list()

for (sigma in 11:50) sigmas <- c(sigmas, sigma/10.0-0.05, sigma/10.0)

for (n in nSize){

for (mean in 0:1){

power = c()

for (sigma in sigmas){

p = 0

for (i in 1:N){

pp = ansari.test(rnorm(n, 0, 1), rnorm(n, 0, sigma), conf.level = 0.95, alternative="greater")$p.value

p = (pp-p)/i + p

}

power = c(power, p)

}

jpeg(paste(n,"_", mean, ".jpg") , width=6, height=6, units="in", res=300)

plot(sigmas,power,pch=19,xlab="variance",ylab="power", main=paste("Mean =", mean, "Sample Size =", n), xlim=c(1,5), ylim=c(0.4,1))

piecewiseModel <- segmented(lm(power~sigmas), seg.Z = ~sigmas, psi = c(2, 3, 4))

par(new = TRUE)

plot(piecewiseModel, col="red", add=TRUE, lwd=4) # ylab="", xlab="", xlim=c(1,5), ylim=c(0.4,1))

dev.off()

fun[[length(fun)+1]] = piecewiseModel

}

}



# Compare graph

jpeg("comparison.jpg" , width=10, height=10, units="in", res=300)

for (i in 1:length(fun)){

if (i%%2 == 1){

if (i == 1){

plot(fun[[i]],xlab="variance",ylab="power", main="Comparison", xlim=c(1,5), ylim=c(0.4,1))

}else{

par(new = TRUE)

plot(fun[[i]], add=TRUE, lwd=2)

}

}else{

par(new = TRUE)

plot(fun[[i]], col="red", add=TRUE, lwd=2)

}

}

dev.off()
